12. see/add details from admin page
create model for complaints:

from datetime import timezone
from django.db import models


class Complaints(models.Model):
    title = models.CharField(max_length=200)
    text = models.TextField()
    priority = models.PositiveIntegerField
    attachment = models.FileField(upload_to='uploads/%Y/%m/%d/')
    created_at = models.DateTimeField(auto_now_add=True)


a.python manage.py makemigrations

c. to see the model in the admin page, you need to import the model in the admin.py file of each app:

from django.contrib import admin
from .models import Complaints
from django import forms


class ComplaintsAdmin(admin.ModelAdmin):
    list_display = ('title', 'text', 'created_at')


admin.site.register(Complaints, ComplaintsAdmin)
